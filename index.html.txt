        function addClient() {
            const name = document.getElementById('newClientName').value.trim();
            const code = document.getElementById('newClientCode').value.trim().toUpperCase();
            if (!name || !code) { showToast('Name and code required!'); return; }
            
            clients[code] = {
                name: name,
                type: document.getElementById('newClientType').value.trim() || 'business',
                services: document.getElementById('newClientServices').value.split('\\n').filter(s=>s.trim()),
                whatsapp: document.getElementById('newClientWhatsApp').value.trim() || globalSettings.whatsapp,
                hours: document.getElementById('newClientHours').value.trim() || 'Mon-Sat 9AM-6PM',
                faqs: {}
            };
            
            const faqText = document.getElementById('newClientFAQs').value.trim();
            if (faqText) {
                faqText.split('\\n').forEach(line => {
                    const parts = line.split('|');
                    if (parts.length === 2) clients[code].faqs[parts[0].trim().toLowerCase()] = parts[1].trim();
                });
            }
            
            showToast('Client "' + name + '" added!');
            clearClientForm();
            renderClients();
        }
        
        function clearClientForm() {
            ['newClientName','newClientType','newClientCode','newClientServices','newClientWhatsApp','newClientHours','newClientFAQs'].forEach(id => document.getElementById(id).value = '');
        }
        
        function renderClients() {
            const list = document.getElementById('clientList');
            list.innerHTML = '';
            Object.entries(clients).forEach(([code, client]) => {
                if (code === 'default') return;
                const div = document.createElement('div');
                div.className = 'client-item';
                div.innerHTML = '<span>' + client.name + ' (' + code + ')</span><div class="client-actions"><button class="edit-btn" onclick="editClient(\'' + code + '\')">Edit</button><button class="del-btn" onclick="deleteClient(\'' + code + '\')">Delete</button></div>';
                list.appendChild(div);
            });
        }
        
        function editClient(code) {
            const c = clients[code];
            if (!c) return;
            document.getElementById('newClientName').value = c.name;
            document.getElementById('newClientType').value = c.type;
            document.getElementById('newClientCode').value = code;
            document.getElementById('newClientServices').value = c.services.join('\\n');
            document.getElementById('newClientWhatsApp').value = c.whatsapp;
            document.getElementById('newClientHours').value = c.hours;
            const faqs = Object.entries(c.faqs).map(([q,a]) => q + ' | ' + a).join('\\n');
            document.getElementById('newClientFAQs').value = faqs;
        }
        
        function deleteClient(code) {

                renderClients();
            }
        }
